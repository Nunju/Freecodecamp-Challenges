function testSize(num) {
  // Only change code below this line
  if (num < 5){
    return "Tiny"
  } else if (num < 10){
    return "Small"
  } else if (num < 15){
    return "Medium"
  } else if (num < 20){
    return "Large"
  } else {
    return "Huge" //pronounced "uge" in Donald Trump's voice with hand motions
  } 

  
  // Only change code above this line
}

testSize(7);
